\section{}
What is the sum of each pair of hexadecimal integers?

\subsection{6B4 + 3FE}

\begin{align*}
	4 + E & = 2, \text{ carry } = 1 \\
	B + F + 1 & = B, \text{ carry } = 1 \\
	6 + 3 + 1 & = A
\end{align*}
$ \text{ 0xAB2 } \equiv 2738 $

\subsection{A49 + 6BD}
\begin{align*}
	9 + D & = 6, \text{ carry = 1 } \\
	4 + B + 1 & = 0, \text{ carry = 1 } \\
	A + 6 + 1 & = 1, \text{ carry = 1 }
\end{align*}
$ \text{ 0x1106 } \equiv 4358 $
